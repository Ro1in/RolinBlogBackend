用户

```sql
CREATE TABLE USER (
	id INT PRIMARY KEY auto_increment,
	username VARCHAR ( 10 ) NOT NULL UNIQUE COMMENT '用户昵称',
	avatar VARCHAR ( 255 ) NULL DEFAULT NULL COMMENT '头像',
	create_date DATETIME NOT NULL COMMENT '注册时间',
	email VARCHAR ( 30 ) NOT NULL UNIQUE COMMENT '邮箱',
	last_login datetime NULL DEFAULT NULL COMMENT '最后登录时间',
	PASSWORD VARCHAR ( 30 ) NOT NULL COMMENT '密码',
	STATUS INT NOT NULL DEFAULT 0 COMMENT '状态'
) ENGINE = INNODB AUTO_INCREMENT=100000 CHARACTER SET = utf8 COLLATE = utf8_general_ci;

ALTER TABLE USER MODIFY COLUMN create_date DATETIME NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间';
```



文章的简略信息

```sql
CREATE TABLE article_abstract (
	id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
	comment_counts INT NULL DEFAULT NULL COMMENT '评论数量',
	create_date datetime NOT NULL COMMENT '创建时间',
	summary VARCHAR ( 255 ) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT '简介',
	title VARCHAR ( 64 ) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT '标题',
	view_counts INT NULL DEFAULT NULL COMMENT '浏览数量',
	top INT NOT NULL COMMENT '是否置顶',
	author_id INT NOT NULL COMMENT '作者id',
	body_id INT NULL DEFAULT NULL COMMENT '内容id',
	category_id INT NULL DEFAULT NULL COMMENT '类别id'
) ENGINE = INNODB CHARACTER SET = utf8 COLLATE = utf8_general_ci;
```

`utf8_general_ci`查询时不区分大小写
